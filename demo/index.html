Simple setup looks like this:

$().image('setup', {srcs: [ 'images/Black_widow.png', 'images/Cap-Shield.png', 'images/Captain-America.png', 'images/Hawkeye.png', 'images/Hulk.png', 'images/Iron-Man.png', 'images/Nick-Fury.png']});

$('#black-widow1, #black-widow2, #black-widow3').image({src: 'images/Black_widow.png'});
$('.cap-shield').image({src: 'images/Cap-Shield.png'});
$('.captain-america').image({src: 'images/Captain-America.png', title: 'Captain America'});
